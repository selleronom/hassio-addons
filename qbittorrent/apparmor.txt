#include <tunables/global>

profile qbittorrent flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  file,
  signal,

  # Capabilities
  capability dac_override,
  capability dac_read_search,
  capability setgid,
  capability setuid,

  # Networking
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,

  # S6-Overlay & Bashio
  /init rix,
  /bin/** ix,
  /usr/bin/** ix,
  /etc/s6-overlay/** rix,
  /etc/s6/** rix,
  /run/s6/** rwix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/** rwk,
  /package/** rix,
  /command/** ix,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rw,

  # Data access
  /data/** rw,

  # Shared storage
  /share/** rw,
  /media/** rw,

  # Config directory
  /config/** rw,

  # qBittorrent binary
  /usr/bin/qbittorrent-nox ix,

  # System files  
  /etc/passwd r,
  /etc/group r,
  /etc/nsswitch.conf r,
  /etc/hosts r,
  /etc/resolv.conf r,
  /etc/ssl/openssl.cnf r,
  /etc/ssl/certs/** r,

  # Proc and sys
  @{PROC}/sys/kernel/random/uuid r,
  /sys/devices/**/net/** r,

  # Deny some dangerous capabilities
  deny capability sys_admin,
  deny capability sys_module,
}
